import React, { PropTypes } from 'react';
import Movie from './movie';

const MovieList = ({ movies }) => {
  return (
    <div className="movie-list">
      {movies.map(movie => <Movie
        key={movie.id}
        movie={movie}
      />)}
    </div>
  );
};

MovieList.propTypes = {
  movies: PropTypes.array.isRequired
};

export default MovieList;
